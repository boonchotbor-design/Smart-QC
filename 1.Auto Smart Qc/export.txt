function doPost(e) {
  const lock = LockService.getScriptLock();
  try {
    if (lock.tryLock(5000)) { // ล็อคป้องกันการทำงานซ้อนกัน
      if (!e.postData || !e.postData.contents) return ContentService.createTextOutput("OK");
      const data = JSON.parse(e.postData.contents);
      
      const cache = CacheService.getScriptCache();
      const uId = data.update_id ? "v103_" + data.update_id : null;
      if (uId && cache.get(uId)) return ContentService.createTextOutput("OK");
      if (uId) cache.put(uId, "1", 3600);

      if (data.events) handleLineWebhook(data.events[0]);
      else handleTelegramWebhook(data);
      
      logToSheet(data);
      lock.releaseLock();
    }
  } catch (err) { console.error(err); }
  return ContentService.createTextOutput("OK");
}

function runPatInspector() {
  const ts = ScriptApp.getProjectTriggers();
  for (let i=0; i<ts.length; i++) { if (ts[i].getHandlerFunction() === 'runPatInspector') ScriptApp.deleteTrigger(ts[i]); }
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(SHEET_NAME) || ss.insertSheet(SHEET_NAME);
  const files = getAllRecursiveFiles(DriveApp.getFolderById(FOLDER_ID));
  const mainArc = DriveApp.getFolderById(ARCHIVE_FOLDER_ID);
  let countP = 0, countF = 0, curS = "Unknown";
  for (let j=0; j<files.length; j++) {
    const f = files[j];
    if (f.getDescription() && f.getDescription().indexOf("PAT_CHECKED") > -1) continue;
    try {
      const siteN = f.getParents().next().getName(); curS = siteN;
      const ai = analyzeAI(f); const status = ai.status.toUpperCase();
      sheet.appendRow([new Date(), f.getName(), ai.sheetReference, status, ai.reason, f.getUrl(), f.getId()]);
      const target = getOrCreateSubFolder(getOrCreateSubFolder(mainArc, siteN), status === "PASS" ? ai.sheetReference : "FAIL_" + ai.sheetReference);
      f.moveTo(target);
      f.setDescription("PAT_CHECKED: " + status + " | " + (f.getDescription() || ""));
      if (status === "PASS") countP++; else sendFailNotify(f.getName(), ai.sheetReference, ai.reason, f.getUrl(), f.getId());
    } catch (e) {}
  }
  if (countP + countF > 0) sendTG(TELEGRAM_TARGET_ID, "📊 <b>สรุปผล AI (" + curS + ")</b>\n✅ ผ่าน: " + countP + "\n❌ ไม่ผ่าน: " + countF, ["ส่งงาน"]);
}
